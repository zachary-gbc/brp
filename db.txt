CREATE DATABASE IF NOT EXISTS brp_prod;

DROP TABLE IF EXISTS ReadingPlanDay;
CREATE TABLE ReadingPlanDay(
  RP_ID INT AUTO_INCREMENT NOT NULL,
  RP_PlanInstance VARCHAR(255) NOT NULL,
  RP_Date INT NOT NULL,
  RP_Verses VARCHAR(255) NOT NULL,
  RP_Version VARCHAR(255) NOT NULL,
  RP_Notes TEXT NULL,
  RP_UpdateDateTime DATETIME NOT NULL,
  PRIMARY KEY(RP_ID)
);

DROP TABLE IF EXISTS Variables;
CREATE TABLE Variables(
  Var_ID INT NOT NULL AUTO_INCREMENT,
  Var_Name VARCHAR(255) NOT NULL,
  Var_Value VARCHAR(255) NULL,
  PRIMARY KEY(Var_ID)
);
